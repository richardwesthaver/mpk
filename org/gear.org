void OTWriter::setChecksum()
{
unsigned short value = 0;
for (int i = 16; i < OTFILESIZE - 2; i++)
value += data[i];
data[OTFILESIZE - 2] = (value >> 8)&0xFF;
data[OTFILESIZE - 1] = value&0xFF;
}
